题目描述：输入一个链表，从尾到头打印链表每个节点的值。

JS
```
/*function ListNode(x){
    this.val = x;
    this.next = null;
}*/
function printListFromTailToHead(head)
{
    // write code here
    var arr = []
    while(head) {
		arr.unshift(head.val)
        head = head.next
    } 
    return arr
}
module.exports = {
    printListFromTailToHead : printListFromTailToHead
};
```

function printListFromTailToHead(head)
{
    // write code here
	if (head !== null) {
        if (head.next === null) {
            return [head.val]
        }else {
            return printListFromTailToHead(head.next).concat(head.val)
        }
    } else {
        return []
    }
}
module.exports = {
    printListFromTailToHead : printListFromTailToHead
};

Java

/**
*    public class ListNode {
*        int val;
*        ListNode next = null;
*
*        ListNode(int val) {
*            this.val = val;
*        }
*    }
*
*/
import java.util.ArrayList;
public class Solution {
    public ArrayList<Integer> printListFromTailToHead(ListNode listNode) {
        ArrayList<Integer> list = new ArrayList<>(1024);
        if (listNode == null)
            return list;
        getVal(listNode, list);
        return list;
    }
    public void getVal(ListNode listNode, ArrayList<Integer> list) {
        if (listNode.next != null) 
            getVal(listNode.next, list);
        list.add(listNode.val);
            
    }
}
